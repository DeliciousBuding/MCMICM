\section{Notations}
\label{sec:notations}

For clarity and consistency throughout this paper, we summarize the key symbols used in our models in Table~\ref{tab:notations}.

\begin{table}[H]
    \centering
    \caption{Summary of Key Notations}
    \label{tab:notations}
    \begin{tabularx}{\textwidth}{c X c}
        \toprule
        \textbf{Symbol} & \textbf{Description} & \textbf{Unit/Range} \\
        \midrule
        $s$ & Season index & $s \in \{1,\ldots,34\}$ \\
        $w$ & Week index within a season & $w \in \{1,\ldots,W_s\}$ \\
        $i$ & Contestant index & -- \\
        $\mathcal{C}_{s,w}$ & Active contestant set in season $s$, week $w$ & -- \\
        $n_{s,w}$ & Number of active contestants, $|\mathcal{C}_{s,w}|$ & -- \\
        \midrule
        $J_{i,w}$ & Judge score for contestant $i$ in week $w$ & [0, 40] \\
        $\bar{J}_w$ & Sum of judge scores in week $w$, $\sum_i J_{i,w}$ & -- \\
        $E_w$ & Eliminated contestant index in week $w$ & -- \\
        $r^{\mathrm{fan}}_{i,w}$ & Fan vote rank (latent in rank seasons) & $\{1,\ldots,n_{s,w}\}$ \\
        \midrule
        $v_{i,w}$ & Fan vote share for contestant $i$ in week $w$ & $[0,1]$, $\sum_i v_{i,w}=1$ \\
        $\hat{v}_{i,w}$ & Posterior mean of fan vote share & $[0,1]$ \\
        $[v^{\min}_{i,w}, v^{\max}_{i,w}]$ & Feasible interval from LP/MILP & $[0,1]$ \\
        $\mathrm{HDI}_{i,w}$ & 95\% highest density interval for $v_{i,w}$ & $[0,1]$ \\
        \midrule
        $\epsilon$ & Minimum vote share floor & $[0,0.1]$ \\
        $S^*$ & Minimum slack (mismatch indicator) & $\ge 0$ \\
        $\lambda$ & Temporal smoothness penalty in MCMC & $>0$ \\
        $\alpha$ & Judge weight in hybrid mechanism & $[0,1]$ \\
        \midrule
        $T_i$ & Survival weeks of contestant $i$ & weeks \\
        $f(\mathbf{x}_i)$ & XGBoost prediction for $T_i$ & weeks \\
        $\phi_j$ & SHAP value for feature $j$ & -- \\
        \midrule
        $J_i^{\%}$ & Judge percentage for contestant $i$ & $[0,1]$ \\
        $S_i^{\mathrm{wp}}$ & Weighted-percent score, $\alpha J_i^{\%} + (1-\alpha)v_i$ & $[0,1]$ \\
        \bottomrule
    \end{tabularx}
\end{table}

\noindent\textbf{Indexing Conventions.}
\begin{itemize}[itemsep=0.2em]
    \item Seasons are numbered chronologically: S1 (2005) through S34 (2025).
    \item Weeks are numbered starting from the first competition episode (excluding any premiere without elimination).
    \item \emph{Percent-rule seasons} (S3--S27): combined scores defined by percentage rule; fan shares unobserved.
    \item \emph{Rank-rule seasons} (S1--S2, S28--S34): only ordinal rankings are revealed.
\end{itemize}
