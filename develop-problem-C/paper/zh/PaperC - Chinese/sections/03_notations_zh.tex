\section{符号说明}
\label{sec:notations}

\begin{table}[H]
\centering
\caption{关键符号汇总}
\label{tab:notations}
\begin{tabularx}{\textwidth}{l X l}
\toprule
\textbf{符号} & \textbf{描述} & \textbf{公式/单位} \\
\midrule
$n$ & 相应周次的活跌选手数量 & 整数 \\
$J_i$ & 选手 $i$ 的原始评委总分 & $[3, 30]$ \\
$J_i^\%$ & 选手 $i$ 的评委分占百分比份额 & $\in [0, 1]$ \\
$R_{J,i}$ & 选手 $i$ 的评委排名（1为最高） & $\in [1, n]$ \\
$v_i$ & 选手 $i$ 的观众投票份额（推断的目标） & $\sum v_i = 1$ \\
$R_{v,i}$ & 选手 $i$ 的观众排名等级 & $\in [1, n]$ \\
$T_i$ & 选手 $i$ 的综合得分（用于决定淘汰） & 取决于规则 \\
$\mathcal{D}_W$ & 确认的非预期淘汰案例集合 & 集合 \\
$s$ & 线性规划中的松弛变量（反映不匹配） & $\ge 0$ \\
$\lambda(t)$ & Cox模型的风险率函数 & 概率密度 \\
$\alpha$ & 机制设计中的评委权重因子 & $\in [0, 1]$ \\
\bottomrule
\end{tabularx}
\end{table}
