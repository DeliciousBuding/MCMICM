\section{模型二：反事实赛制评估}
\label{sec:model2_zh}

\noindent\textbf{\textit{对应问题 2：}} 固定粉丝意愿与评委分数，替换规则权重，评估赛制公平性与稳定性。

\subsection{评估指标}
\begin{itemize}[itemsep=0.2em]
    \item \textbf{Kendall 相关}：排名一致性。
    \item \textbf{逆转率}：机制淘汰是否与观众淘汰不同。
    \item \textbf{孔多塞一致性}：是否淘汰粉丝赢家。
\end{itemize}

\subsection{结果}
\begin{table}[H]
    \centering
    \caption{反事实评估汇总}
    \label{tab:counterfactual_summary_zh}
    \begin{tabular}{lccc}
        \toprule
        \textbf{规则} & \textbf{Kendall 相关} & \textbf{逆转率} & \textbf{孔多塞一致性} \\
        \midrule
        固定 60/40（$\alpha=0.6$） & 0.455 & 0.174 & 1.000 \\
        动态权重 & 0.409 & 0.174 & 1.000 \\
        \bottomrule
    \end{tabular}
\end{table}

\noindent\textbf{解读：} 动态权重略降低 Kendall 相关（更偏观众波动），但逆转率与孔多塞一致性保持不变。
