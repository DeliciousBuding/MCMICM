\section{Model Evaluation and Further Discussion}
\label{sec:evaluation}

% 8.1 模型优势
\subsection{Strengths}
\label{subsec:strengths}

\begin{enumerate}[itemsep=0.4em]
    \item \textbf{Hybrid Architecture:} The PCA--LSTM--XGBoost pipeline effectively combines dimensionality reduction, temporal modeling, and non-linear regression, capturing complex patterns in Olympic data.
    
    \item \textbf{Uncertainty Quantification:} Bootstrap-based 95\% confidence intervals provide actionable uncertainty bounds, enabling risk-aware decision-making for NOCs.
    
    \item \textbf{Causal Identification:} The DID framework with parallel-trends validation offers credible causal inference for coaching interventions, moving beyond mere correlation.
    
    \item \textbf{Zero-Inflation Handling:} The Hurdle--Tobit model appropriately addresses the prevalence of zero-medal observations among smaller NOCs.
    
    \item \textbf{Interpretability:} SHAP-based explanations demystify model predictions, enhancing trust and actionability for stakeholders.
\end{enumerate}

% 8.2 模型局限
\subsection{Weaknesses}
\label{subsec:weaknesses}

\begin{enumerate}[itemsep=0.4em]
    \item \textbf{Data Limitations:} Historical records before 1950 are sparse and may introduce measurement error; some NOCs have very few observations.
    
    \item \textbf{Computational Cost:} The LSTM training and bootstrap procedure require substantial computation, potentially limiting real-time updates.
    
    \item \textbf{Unobserved Confounders:} Despite our controls, unmeasured factors (e.g., doping scandals, geopolitical events) may bias causal estimates.
    
    \item \textbf{Generalization:} The model is calibrated on Summer Olympics; extension to Winter Olympics or other multi-sport events would require retraining.
\end{enumerate}

% 8.3 进一步讨论
\subsection{Further Discussion}
\label{subsec:further}

\subsubsection{Potential Improvements}
\begin{itemize}[itemsep=0.2em]
    \item \textbf{Attention Mechanisms:} Replacing LSTM with Transformer-based architectures may capture long-range dependencies more effectively.
    \item \textbf{Synthetic Control:} For cases with few treated units, synthetic control methods could complement DID.
    \item \textbf{Multi-Task Learning:} Jointly predicting gold and total medals may improve efficiency through shared representations.
\end{itemize}

\subsubsection{Deployment Considerations}
\begin{itemize}[itemsep=0.2em]
    \item \textbf{Data Pipeline:} Automated ingestion of post-Games results would enable continuous model updating.
    \item \textbf{Dashboard Integration:} Interactive visualizations could help NOC administrators explore ``what-if'' scenarios.
    \item \textbf{Ethical Guidelines:} Predictions should not be used to reduce funding for historically disadvantaged nations; equity considerations matter.
\end{itemize}
