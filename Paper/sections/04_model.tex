\subsection{Solution Algorithm}
% 提示：算法伪代码不仅显得专业，也是填充页面长度的艺术。
\begin{algorithm}[H]
    \caption{The Optimization Process}
    \KwIn{Raw dataset $\mathcal{D}$, iteration limit $N$}
    \KwOut{Optimized parameters $\Theta^*$}
    Initialize $\Theta_0$ randomly\;
    \For{$k=1$ \KwTo $N$}{
        Calculate gradient $\nabla J(\Theta_k)$\;
        Update $\Theta_{k+1} = \Theta_k - \eta \nabla J$\;
    }
    \Return{$\Theta_{N}$}\;
\end{algorithm}
